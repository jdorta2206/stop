// components/game/stop-button.tsx
'use client';

import { Button } from "@/components/ui/button";

// Definición segura del tipo
type LanguageCode = 'es' | 'en' | 'fr' | 'pt';

const TEXT_MAP: Record<LanguageCode, string> = {
  es: 'DETENER',
  en: 'STOP',
  fr: 'ARRÊTER',
  pt: 'PARAR'
};

interface StopButtonProps {
  onClick: () => void;
  disabled: boolean;
  language: LanguageCode;
}

export function StopButton({ onClick, disabled, language }: StopButtonProps) {
  return (
    <Button
      onClick={onClick}
      disabled={disabled}
      variant="destructive"
      size="lg"
      className="text-xl font-bold px-10 py-6 rounded-full shadow-lg hover:bg-red-600 transition-colors animate-pulse"
    >
      {TEXT_MAP[language]}
    </Button>
  );
}
